    <script>
        // 初始化 Lucide 圖標
        lucide.createIcons();

        // 頁面視圖控制邏輯
        const views = ['authView', 'projectView', 'formView'];
        
        function showView(viewId) {
            views.forEach(v => document.getElementById(v).classList.add('hidden'));
            document.getElementById(viewId).classList.remove('hidden');
            window.scrollTo(0, 0);
        }

        function toggleAuth(isLogin) {
            document.getElementById('loginForm').classList.toggle('hidden', !isLogin);
            document.getElementById('registerForm').classList.toggle('hidden', isLogin);
        }

        // 模擬動作：登入
        function login() {
            document.getElementById('userNav').classList.remove('hidden');
            showView('projectView');
        }

        // 模擬動作：登出
        function logout() {
            document.getElementById('userNav').classList.add('hidden');
            showView('authView');
        }

        // 模擬動作：發送驗證信
        function sendVerification() {
            showModal('mail', '驗證信已寄出', '請至您的電子信箱點擊驗證連結，以啟動您的帳號。', () => {
                toggleAuth(true);
            });
        }

        // 進入投件表單
        function startApplication(title) {
            document.getElementById('currentProjectTitle').innerText = title;
            showView('formView');
        }

        function goToProjects() {
            showView('projectView');
        }

        // 表單經費計算
        function calculateTotal() {
            const grant = parseFloat(document.getElementById('grantAmount').value) || 0;
            const self = parseFloat(document.getElementById('selfAmount').value) || 0;
            document.getElementById('totalAmount').innerText = (grant + self).toLocaleString();
        }

        // 檔案處理
        function triggerUpload(id) {
            document.getElementById(id).click();
        }

        function handleFile(input, statusId) {
            if (input.files && input.files[0]) {
                const file = input.files[0];
                const sizeInMB = (file.size / (1024 * 1024)).toFixed(2);
                
                // 1GB 限制檢查 (1024MB)
                if (file.size > 1024 * 1024 * 1024) {
                    showModal('alert-triangle', '檔案超過限制', '單一檔案大小不能超過 1GB，請重新調整。', null);
                    input.value = "";
                    return;
                }
                
                const statusEl = document.getElementById(statusId);
                statusEl.innerText = `${file.name} (${sizeInMB} MB)`;
                statusEl.classList.replace('text-slate-500', 'text-indigo-600');
                statusEl.classList.replace('bg-slate-100', 'bg-indigo-50');
            }
        }

        // 暫存草稿
        function saveDraft() {
            showModal('save', '暫存成功', '您的資料已成功儲存，稍後可登入繼續編輯。', null);
        }

        // 提交計畫 (模擬進度條)
        function submitProject() {
            const progressArea = document.getElementById('uploadProgress');
            const submitBtn = document.getElementById('submitBtn');
            const bar = document.getElementById('progressBar');
            const label = document.getElementById('percentLabel');
            
            progressArea.classList.remove('hidden');
            submitBtn.disabled = true;
            submitBtn.innerText = "正在加密上傳中...";
            submitBtn.classList.add('opacity-50');

            let progress = 0;
            const interval = setInterval(() => {
                progress += Math.random() * 12; // 隨機增加進度
                if (progress >= 100) {
                    progress = 100;
                    clearInterval(interval);
                    
                    showModal('check-circle', '提交成功', '計畫書已正式送出。系統已同步發送收件編號至您的信箱。', () => {
                        showView('projectView');
                        // 重置按鈕與進度條
                        progressArea.classList.add('hidden');
                        submitBtn.disabled = false;
                        submitBtn.innerText = "確認送出計畫書";
                        submitBtn.classList.remove('opacity-50');
                        bar.style.width = '0%';
                        label.innerText = '0%';
                    });
                }
                bar.style.width = progress + '%';
                label.innerText = Math.floor(progress) + '%';
            }, 300);
        }

        // 自定義彈窗邏輯
        let modalCallback = null;
        function showModal(icon, title, body, callback) {
            document.getElementById('modalTitle').innerText = title;
            document.getElementById('modalBody').innerText = body;
            const iconDiv = document.getElementById('modalIcon');
            
            // 替換 Lucide 圖標
            iconDiv.innerHTML = `<i data-lucide="${icon}" class="w-8 h-8"></i>`;
            lucide.createIcons();
            
            document.getElementById('messageModal').style.display = 'flex';
            modalCallback = callback;
        }

        function closeModal() {
            document.getElementById('messageModal').style.display = 'none';
            if (modalCallback) modalCallback();
        }
